class Program
{
	try
	{
    		int number;
   	 	Console.Write("Enter Number :");
    		number = Convert.ToInt32(Console.ReadLine());
    		Console.WriteLine(Reverse_a_given_number(number));
    		static int Reverse_a_given_number(int number)
    		{
        		int remainder, sum = 0;
        		while (number != 0)
        		{
            			remainder = number % 10;
            			sum = sum * 10 + remainder;
            			number /= 10;
        		}
        		return sum;
    		}
	}
	catch(FormatException ex)
	{
    		Console.WriteLine("Invalid input");
    		Console.WriteLine("Please enter Integer number");
	}
}
